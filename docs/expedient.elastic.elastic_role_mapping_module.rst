Examples
--------

.. code-block:: yaml

    - name: Elastic Role Mapping
      expedient.elastic.elastic_role_mapping:
        host: '{{ elastic_host }}'
        username: '{{ elastic_username }}'
        password: '{{ elastic_password }}'
        state: present
        name: Example
        enabled: yes
        roles:
        - example_list
        rules:
          example_rule: need_real_data_here
        metadata:
          example_metadata: need_real_data_here
